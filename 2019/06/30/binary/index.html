.data

start:
        mov ax, @data
        mov ds, ax

        lea dx, str
        mov ah, 10
        int 21h

        mov cx, str+1
        lea bx, word ptr str

loop1:
        mov al, [bx]
        cmp al, 'a'
        jb next1
        cmp al, 'z'
        ja next1
        inc alpha
        jmp exit

next1:
        cmp al, 'A'
        jb next2
        cmp al, 'Z'
        ja next2
        inc alpha
        jmp exit

next2:
        cmp al, '0'
        jb next3
        cmp al, '9'
        ja next3
        inc digit
        jmp exit

next3:
        inc other

exit:
        inc bx
        loop loop1

        mov ax, 4c00h
        int 21h
end start

COUNT DB $-ARRAY

MAXAY DW ? ;存放最大值

MINAY DW ? ;存放最小值

绘制程序的流程图，写出这个程序

start:
    mov ax, @data
    mov ds, ax
    mov ax, count
    shr ax, 1
    mov cx, ax
    mov si, 0
    mov MAXAY, 0
    mov MINAY, FFh

next:
    mov bx, MAXAY
    cmp bx, appay[si]
    jnb max1
    mov bx, MINAY
    cpm bx, appay[si]
    jna min1

max1:
    mov ax,array[si]
    mov MAXAY, ax

min1:
    mov ax,array[si]
    mov MINAY, ax

    loop next
    mov ah, 4ch
    int 21h
end start
